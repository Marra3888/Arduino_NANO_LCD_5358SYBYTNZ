#include <avr/pgmspace.h>

   Font tables
   ============================================================ */

const uint8_t GLYPH_DOT  [5] PROGMEM = {0x00,0x60,0x60,0x00,0x00};
const uint8_t GLYPH_COLON[5] PROGMEM = {0x00,0x36,0x36,0x00,0x00};
const uint8_t GLYPH_SLASH[5] PROGMEM = {0x20,0x10,0x08,0x04,0x02};

const uint8_t GLYPH_0[5] PROGMEM = {0x3E,0x51,0x49,0x45,0x3E};
const uint8_t GLYPH_1[5] PROGMEM = {0x00,0x42,0x7F,0x40,0x00};

const uint8_t GLYPH_W[5] PROGMEM = {0x7F,0x20,0x18,0x20,0x7F};
const uint8_t GLYPH_X[5] PROGMEM = {0x63,0x14,0x08,0x14,0x63};
const uint8_t GLYPH_Y[5] PROGMEM = {0x03,0x04,0x78,0x04,0x03};
const uint8_t GLYPH_Z[5] PROGMEM = {0x61,0x51,0x49,0x45,0x43};

/* ============================================================
   CustomLCD class
   ============================================================ */
class CustomLCD
{
public:
  void begin()
  {
    pinMode(LCD_PIN_A0X, OUTPUT);
    pinMode(LCD_PIN_MODEX, OUTPUT);
    pinMode(LCD_PIN_STBX, OUTPUT);
    pinMode(LCD_PIN_CS2X, OUTPUT);
    pinMode(LCD_PIN_RSTX, OUTPUT);

    for (uint8_t i = 0; i < 8; i++)
    {
      pinMode(lcdDb[i], OUTPUT);
      digitalWrite(lcdDb[i], LOW);
    }

    modeIdle();
    setCS2(false);
    setA0(false);
    stbIdle();
    setRST(false);

    delay(50);

    init();
    clearAllHw();
    clearTextBuffers();
    clearSpecialBuffers();
    update();
  }

  void clear()
  {
    clearTextBuffers();
    clearSpecialBuffers();
    update();
  }

  void setText(
    const char* leftTop, const char* rightTop,
    const char* leftBottom, const char* rightBottom)
  {
    clearTextBuffers();

    renderTextToBuffer(leftTextFb[0], LEFT_TEXT_COLS, leftTop, LEFT_TEXT_CHARS);
    renderTextToBuffer(rightTextFb[0], RIGHT_TEXT_COLS, rightTop, RIGHT_TEXT_CHARS);

    renderTextToBuffer(leftTextFb[1], LEFT_TEXT_COLS, leftBottom, LEFT_TEXT_CHARS);
    renderTextToBuffer(rightTextFb[1], RIGHT_TEXT_COLS, rightBottom, RIGHT_TEXT_CHARS);
  }

  void show(
    const char* leftTop, const char* rightTop,
    const char* leftBottom, const char* rightBottom)
  {
    setText(leftTop, rightTop, leftBottom, rightBottom);
    update();
  }

  void setLeftTriangle1(bool on) { setBit(leftSpecFb[0], 4, on); }
  void setLeftTriangle2(bool on) { setBit(leftSpecFb[0], 7, on); }
  void setLeftTriangle3(bool on) { setBit(leftSpecFb[1], 1, on); }
  void setLeftTriangle4(bool on) { setBit(leftSpecFb[1], 3, on); }

  void setLeftTriangles(bool t1, bool t2, bool t3, bool t4)
  {
    setLeftTriangle1(t1);
    setLeftTriangle2(t2);
    setLeftTriangle3(t3);
    setLeftTriangle4(t4);
  }

  void setA(bool on) { setBit(rightAbcdFb[0], 5, on); }
  void setB(bool on) { setBit(rightAbcdFb[0], 6, on); }
  void setC(bool on) { setBit(rightAbcdFb[1], 1, on); }
  void setD(bool on) { setBit(rightAbcdFb[1], 2, on); }

  void setABCD(bool a, bool b, bool c, bool d)
  {
    setA(a); setB(b); setC(c); setD(d);
  }

  void setR(bool on) { setBit(rightRsxzFb[0], 5, on); }
  void setS(bool on) { setBit(rightRsxzFb[0], 6, on); }
  void setX(bool on) { setBit(rightRsxzFb[1], 1, on); }
  void setZ(bool on) { setBit(rightRsxzFb[1], 2, on); }

  void setRSXZ(bool r, bool s, bool x, bool z)
  {
    setR(r); setS(s); setX(x); setZ(z);
  }

  void setRightTriangle1(bool on) { setBit(rightTriFb[0], 5, on); }
  void setRightTriangle2(bool on) { setBit(rightTriFb[0], 6, on); }
  void setRightTriangle3(bool on) { setBit(rightTriFb[1], 1, on); }
  void setRightTriangle4(bool on) { setBit(rightTriFb[1], 2, on); }

  void setRightTriangles(bool t1, bool t2, bool t3, bool t4)
  {
    setRightTriangle1(t1);
    setRightTriangle2(t2);
    setRightTriangle3(t3);
    setRightTriangle4(t4);
  }

  void update()
  {
    for (uint8_t page = 0; page < LCD_PAGES; page++)
    {
      setPageColRaw(PHYS_CHIP_SCREEN_RIGHT, page, SCREEN_RIGHT_TEXT_START_COL);
      for (uint8_t x = 0; x < RIGHT_TEXT_COLS; x++)
        data(PHYS_CHIP_SCREEN_RIGHT, rightTextFb[page][x]);

      setPageColRaw(PHYS_CHIP_SCREEN_RIGHT, page, SCREEN_RIGHT_SPECIAL_COL_ABCD);
      data(PHYS_CHIP_SCREEN_RIGHT, rightAbcdFb[page]);

      setPageColRaw(PHYS_CHIP_SCREEN_RIGHT, page, SCREEN_RIGHT_SPECIAL_COL_RSXZ);
      data(PHYS_CHIP_SCREEN_RIGHT, rightRsxzFb[page]);

      setPageColRaw(PHYS_CHIP_SCREEN_RIGHT, page, SCREEN_RIGHT_SPECIAL_COL_TRI);
      data(PHYS_CHIP_SCREEN_RIGHT, rightTriFb[page]);

      delayMicroseconds(PAGE_PAUSE_US);

      setPageColRaw(PHYS_CHIP_SCREEN_LEFT, page, SCREEN_LEFT_TEXT_START_COL);
      for (uint8_t x = 0; x < LEFT_TEXT_COLS; x++)
        data(PHYS_CHIP_SCREEN_LEFT, leftTextFb[page][x]);

      setPageColRaw(PHYS_CHIP_SCREEN_LEFT, page, SCREEN_LEFT_SPECIAL_COL);
      data(PHYS_CHIP_SCREEN_LEFT, leftSpecFb[page]);

      delayMicroseconds(PAGE_PAUSE_US);
    }
  }

private:
  /* pins */
  const uint8_t LCD_PIN_A0X   = 10;
  const uint8_t LCD_PIN_MODEX = 11;
  const uint8_t LCD_PIN_STBX  = 12;
  const uint8_t LCD_PIN_CS2X  = A1;
  const uint8_t LCD_PIN_RSTX  = A2;

  const uint8_t lcdDb[8] = {2,3,4,5,6,7,8,9};

  /* geometry */
  static const uint8_t LCD_PAGES = 2;
  static const uint8_t LCD_RAM_COLS = 80;
  static const uint8_t LEFT_TEXT_COLS = 60;
  static const uint8_t RIGHT_TEXT_COLS = 72;
  static const uint8_t LEFT_TEXT_CHARS = 10;
  static const uint8_t RIGHT_TEXT_CHARS = 12;

  static const uint8_t CHIP_LEFT  = 0;
  static const uint8_t CHIP_RIGHT = 1;

  static const bool LEFT_CHIP_IS_CS2 = true;

  static const uint8_t PHYS_CHIP_SCREEN_LEFT  = CHIP_RIGHT;
  static const uint8_t PHYS_CHIP_SCREEN_RIGHT = CHIP_LEFT;

  static const uint8_t SCREEN_LEFT_TEXT_START_COL   = 0;
  static const uint8_t SCREEN_LEFT_SPECIAL_COL      = 60;
  static const uint8_t SCREEN_RIGHT_TEXT_START_COL  = 0;
  static const uint8_t SCREEN_RIGHT_SPECIAL_COL_ABCD = 74;
  static const uint8_t SCREEN_RIGHT_SPECIAL_COL_RSXZ = 75;
  static const uint8_t SCREEN_RIGHT_SPECIAL_COL_TRI  = 76;

  static const bool RST_ACTIVE_HIGH = true;
  static const bool A0_DATA_HIGH    = true;
  static const bool STB_ACTIVE_LOW  = true;
  static const bool CS2_ACTIVE_LOW  = true;

  static const bool LCD_ADC_REVERSED = false;
  static const bool FONT_LSB_TOP     = true;

  static const uint8_t CMD_DISPLAY_OFF        = 0xAE;
  static const uint8_t CMD_DISPLAY_ON         = 0xAF;
  static const uint8_t CMD_DISPLAY_START_LINE = 0xC0;
  static const uint8_t CMD_SET_PAGE           = 0xB8;
  static const uint8_t CMD_SET_COLUMN         = 0x00;
  static const uint8_t CMD_ADC_FORWARD        = 0xA0;
  static const uint8_t CMD_ADC_REVERSE        = 0xA1;
  static const uint8_t CMD_STATIC_DRIVE_OFF   = 0xA4;
  static const uint8_t CMD_DUTY_16            = 0xA8;
  static const uint8_t CMD_RESET              = 0xE2;

  static const uint16_t T_SETUP_US = 2;
  static const uint16_t T_PULSE_US = 2;
  static const uint16_t T_HOLD_US  = 2;
  static const uint16_t PAGE_PAUSE_US = 200;

  uint8_t leftTextFb[LCD_PAGES][LEFT_TEXT_COLS];
  uint8_t rightTextFb[LCD_PAGES][RIGHT_TEXT_COLS];

  uint8_t leftSpecFb[LCD_PAGES];
  uint8_t rightAbcdFb[LCD_PAGES];
  uint8_t rightRsxzFb[LCD_PAGES];
  uint8_t rightTriFb[LCD_PAGES];

  void clearTextBuffers()
  {
    for (uint8_t p = 0; p < LCD_PAGES; p++)
    {
      for (uint8_t i = 0; i < LEFT_TEXT_COLS; i++)  leftTextFb[p][i] = 0x00;
      for (uint8_t i = 0; i < RIGHT_TEXT_COLS; i++) rightTextFb[p][i] = 0x00;
    }
  }

  void clearSpecialBuffers()
  {
    for (uint8_t p = 0; p < LCD_PAGES; p++)
    {
      leftSpecFb[p] = 0x00;
      rightAbcdFb[p] = 0x00;
      rightRsxzFb[p] = 0x00;
      rightTriFb[p] = 0x00;
    }
  }

  void clearAllHw()
  {
    clearChipHw(CHIP_LEFT);
    clearChipHw(CHIP_RIGHT);
  }

  void setBit(uint8_t &value, uint8_t bit, bool on)
  {
    if (on) value |= (1 << bit);
    else    value &= ~(1 << bit);
  }

  void pinW(uint8_t pin, bool level)
  {
    digitalWrite(pin, level ? HIGH : LOW);
  }

  bool activeToLevel(bool active, bool activeLow)
  {
    return activeLow ? !active : active;
  }

  bool chipNeedsCs2Active(uint8_t chip)
  {
    if (LEFT_CHIP_IS_CS2) return (chip == CHIP_LEFT);
    return (chip == CHIP_RIGHT);
  }

  void setA0(bool isData)
  {
    pinW(LCD_PIN_A0X, A0_DATA_HIGH ? isData : !isData);
  }

  void setCS2(bool active)
  {
    pinW(LCD_PIN_CS2X, activeToLevel(active, CS2_ACTIVE_LOW));
  }

  void setRST(bool active)
  {
    pinW(LCD_PIN_RSTX, RST_ACTIVE_HIGH ? active : !active);
  }

  void modeIdle()
  {
    pinW(LCD_PIN_MODEX, HIGH);
  }

  void modeWriteBegin()
  {
    pinW(LCD_PIN_MODEX, LOW);
  }

  void modeWriteEnd()
  {
    pinW(LCD_PIN_MODEX, HIGH);
  }

  void stbIdle()
  {
    pinW(LCD_PIN_STBX, activeToLevel(false, STB_ACTIVE_LOW));
  }

  void pulseStrobe()
  {
    pinW(LCD_PIN_STBX, activeToLevel(true, STB_ACTIVE_LOW));
    delayMicroseconds(T_PULSE_US);
    pinW(LCD_PIN_STBX, activeToLevel(false, STB_ACTIVE_LOW));
  }

  void setDataBus(uint8_t value)
  {
    for (uint8_t i = 0; i < 8; i++)
      pinW(lcdDb[i], (value >> i) & 1);
  }

  void writeByte(uint8_t chip, bool isData, uint8_t value)
  {
    setA0(isData);
    setDataBus(value);

    modeIdle();
    delayMicroseconds(1);

    modeWriteBegin();
    setCS2(chipNeedsCs2Active(chip));

    delayMicroseconds(T_SETUP_US);
    pulseStrobe();

    setCS2(false);
    modeWriteEnd();

    delayMicroseconds(T_HOLD_US);
  }

  void cmd(uint8_t chip, uint8_t value)
  {
    writeByte(chip, false, value);
  }

  void data(uint8_t chip, uint8_t value)
  {
    writeByte(chip, true, value);
  }

  void resetHw()
  {
    modeIdle();
    setCS2(false);
    setA0(false);
    stbIdle();

    setRST(true);
    delay(30);
    setRST(false);
    delay(60);
  }

  void setPageColRaw(uint8_t chip, uint8_t page, uint8_t col)
  {
    if (col >= LCD_RAM_COLS) col = LCD_RAM_COLS - 1;
    cmd(chip, CMD_SET_PAGE | (page & 0x03));
    cmd(chip, CMD_SET_COLUMN | (col & 0x7F));
  }

  void initChip(uint8_t chip)
  {
    cmd(chip, CMD_DISPLAY_OFF);        delay(4);
    cmd(chip, CMD_RESET);              delay(4);

    cmd(chip, CMD_SET_PAGE | 0x00);    delay(4);
    cmd(chip, CMD_SET_COLUMN | 0x00);  delay(4);

    cmd(chip, CMD_DUTY_16);            delay(4);

    if (LCD_ADC_REVERSED) cmd(chip, CMD_ADC_REVERSE);
    else                  cmd(chip, CMD_ADC_FORWARD);
    delay(4);

    cmd(chip, CMD_STATIC_DRIVE_OFF);   delay(4);
    cmd(chip, CMD_DISPLAY_START_LINE); delay(4);
    cmd(chip, CMD_DISPLAY_ON);         delay(4);

    cmd(chip, CMD_SET_PAGE | 0x00);    delay(4);
    cmd(chip, CMD_SET_COLUMN | 0x00);  delay(4);
  }

  void init()
  {
    resetHw();
    initChip(CHIP_LEFT);
    initChip(CHIP_RIGHT);
  }

  void clearChipHw(uint8_t chip)
  {
    for (uint8_t page = 0; page < LCD_PAGES; page++)
    {
      setPageColRaw(chip, page, 0);
      for (uint8_t col = 0; col < LCD_RAM_COLS; col++)
        data(chip, 0x00);

      delayMicroseconds(PAGE_PAUSE_US);
    }
  }

  void renderTextToBuffer(uint8_t *dst, uint8_t cols, const char* s, uint8_t maxChars)
  {
    for (uint8_t i = 0; i < cols; i++) dst[i] = 0x00;

    uint8_t x = 0;
    uint8_t chars = 0;

    while (*s && chars < maxChars && x < cols)
    {
      const uint8_t* g = glyph(*s);

      for (uint8_t i = 0; i < 6; i++)
      {
        uint8_t b = 0x00;
        if (i < 5) b = pgm_read_byte(g + i);
        b = orientGlyphByte(b);

        if (x >= cols) break;
        dst[x++] = b;
      }

      s++;
      chars++;
    }
  }

  const uint8_t* glyph(char c)
  {
    if (c >= 'a' && c <= 'z') c -= 32;

    switch (c)
    {
      case ' ': return GLYPH_SPACE;
      case '-': return GLYPH_DASH;
      case '.': return GLYPH_DOT;
      case ':': return GLYPH_COLON;
      case '/': return GLYPH_SLASH;

      case '0': return GLYPH_0;
      case '1': return GLYPH_1;
      case '2': return GLYPH_2;
      case '3': return GLYPH_3;
      case '4': return GLYPH_4;
      case '5': return GLYPH_5;
      case '6': return GLYPH_6;
      case '7': return GLYPH_7;
      case '8': return GLYPH_8;
      case '9': return GLYPH_9;

      case 'A': return GLYPH_A;
      case 'B': return GLYPH_B;
      case 'C': return GLYPH_C;
      case 'D': return GLYPH_D;
      case 'E': return GLYPH_E;
      case 'F': return GLYPH_F;
      case 'G': return GLYPH_G;
      case 'H': return GLYPH_H;
      case 'I': return GLYPH_I;
      case 'J': return GLYPH_J;
      case 'K': return GLYPH_K;
      case 'L': return GLYPH_L;
      case 'M': return GLYPH_M;
      case 'N': return GLYPH_N;
      case 'O': return GLYPH_O;
      case 'P': return GLYPH_P;
      case 'Q': return GLYPH_Q;
      case 'R': return GLYPH_R;
      case 'S': return GLYPH_S;
      case 'T': return GLYPH_T;
      case 'U': return GLYPH_U;
      case 'V': return GLYPH_V;
      case 'W': return GLYPH_W;
      case 'X': return GLYPH_X;
      case 'Y': return GLYPH_Y;
      case 'Z': return GLYPH_Z;
    }

    return GLYPH_SPACE;
  }

  uint8_t orientGlyphByte(uint8_t b)
  {
    if (FONT_LSB_TOP) return b;

    uint8_t r = 0;
    for (uint8_t i = 0; i < 7; i++)
      if (b & (1 << i))
        r |= (1 << (6 - i));
    return r;
  }
};

/* ============================================================
   Example
   ============================================================ */
CustomLCD lcd;

void setup()
{
  lcd.begin();

  lcd.show(
    "1234567890",
    "ABCDEFGHIJKL",
    "MNOPQRSTUV",
    "WXYZ01234567"
  );

  lcd.setABCD(true, false, true, false);
  lcd.setRSXZ(false, true, false, true);
  lcd.setLeftTriangles(true, false, true, false);
  lcd.setRightTriangles(false, true, false, true);

  lcd.update();
}

void loop()
{
}
