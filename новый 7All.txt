#include <Arduino.h>
#include <avr/pgmspace.h>

/*
  SED1520-like LCD, 134x16
  ========================

  Arduino Nano wiring:

    D2  -> CN10-9   (D0)
    D3  -> CN10-10  (D1)
    D4  -> CN10-11  (D2)
    D5  -> CN10-12  (D3)
    D6  -> CN10-13  (D4)
    D7  -> CN10-14  (D5)
    D8  -> CN10-15  (D6)
    D9  -> CN10-16  (D7)

    D10 -> CN10-4   (A0)
    D11 -> CN10-5   (write gate / mode)
    D12 -> CN10-8   (main write strobe)
    A1  -> CN10-18  (CS2)
    A2  -> CN10-17  (RST)

  Модель по последним VCD:
    - CN10-8 = главный strobe
    - CN10-5 = write-window line: LOW только во время записи
    - CN10-18 = CS2
    - CN10-17 = RST active HIGH
    - 134x16 visible
    - 2 pages
    - 67 bytes per page per chip

  Если половины окажутся перепутаны:
    поменяй LEFT_CHIP_IS_CS2 с 1 на 0
*/

/* ============================================================
   Commands
   ============================================================ */
#define CMD_DISPLAY_OFF          0xAE
#define CMD_DISPLAY_ON           0xAF
#define CMD_DISPLAY_START_LINE   0xC0
#define CMD_SET_PAGE             0xB8
#define CMD_SET_COLUMN           0x00
#define CMD_ADC_FORWARD          0xA0
#define CMD_ADC_REVERSE          0xA1
#define CMD_STATIC_DRIVE_OFF     0xA4
#define CMD_DUTY_16              0xA8
#define CMD_RESET                0xE2

/* ============================================================
   Pins
   ============================================================ */
const uint8_t LCD_DB[8] = {2,3,4,5,6,7,8,9};

#define LCD_PIN_A0    10   // CN10-4
#define LCD_PIN_MODE  11   // CN10-5
#define LCD_PIN_STB   12   // CN10-8
#define LCD_PIN_CS2   A1   // CN10-18
#define LCD_PIN_RST   A2   // CN10-17

/* ============================================================
   Geometry
   ============================================================ */
#define CHIP_MAIN_COLS       67
#define LCD_WIDTH            134
#define LCD_HEIGHT           16
#define LCD_PAGES            2
#define LCD_HALF_WIDTH       CHIP_MAIN_COLS
#define LCD_RAM_COLS         80
#define LCD_CHARS_PER_LINE   22   // 22*6 = 132 px

/* ============================================================
   Settings
   ============================================================ */
#define RST_ACTIVE_HIGH      1
#define A0_DATA_HIGH         1
#define STB_ACTIVE_LOW       1
#define CS2_ACTIVE_LOW       1

// Если текст окажется зеркальным по X
#define LCD_ADC_REVERSED     0

// Если буквы в строке окажутся перевёрнуты по вертикали
#define FONT_LSB_TOP         1

/*
  Какая половина выбирается активным CS2:
    1 = LEFT chip -> CS2 active
    0 = RIGHT chip -> CS2 active
*/
#define LEFT_CHIP_IS_CS2     1

/*
  Смещение видимой области, если потребуется
*/
#define COL_START_LEFT       0
#define COL_START_RIGHT      0

/* ============================================================
   Timing
   ============================================================ */
#define T_SETUP_US           2
#define T_PULSE_US           2
#define T_HOLD_US            2
#define PAGE_PAUSE_US        200

/* ============================================================
   Logical chips
   ============================================================ */
#define CHIP_LEFT            0
#define CHIP_RIGHT           1

/* ============================================================
   Framebuffer
   ============================================================ */
static uint8_t lcd_fb[LCD_PAGES][LCD_WIDTH];

/* ============================================================
   Low-level helpers
   ============================================================ */
static inline void pinW(uint8_t pin, bool level)
{
  digitalWrite(pin, level ? HIGH : LOW);
}

static inline bool activeToLevel(bool active, bool activeLow)
{
  return activeLow ? !active : active;
}

static inline void setA0(bool isData)
{
  pinW(LCD_PIN_A0, A0_DATA_HIGH ? isData : !isData);
}

static inline void setCS2(bool active)
{
  pinW(LCD_PIN_CS2, activeToLevel(active, CS2_ACTIVE_LOW));
}

static inline void setRST(bool active)
{
  pinW(LCD_PIN_RST, RST_ACTIVE_HIGH ? active : !active);
}

static inline void modeIdle(void)
{
  // CN10-5 idle HIGH
  pinW(LCD_PIN_MODE, HIGH);
}

static inline void modeWriteWindowBegin(void)
{
  // CN10-5 LOW only during write
  pinW(LCD_PIN_MODE, LOW);
}

static inline void modeWriteWindowEnd(void)
{
  pinW(LCD_PIN_MODE, HIGH);
}

static inline void stbIdle(void)
{
  pinW(LCD_PIN_STB, activeToLevel(false, STB_ACTIVE_LOW));
}

static inline void pulseStrobe(void)
{
  pinW(LCD_PIN_STB, activeToLevel(true, STB_ACTIVE_LOW));
  delayMicroseconds(T_PULSE_US);
  pinW(LCD_PIN_STB, activeToLevel(false, STB_ACTIVE_LOW));
}

static inline void setDataBus(uint8_t value)
{
  for (uint8_t i = 0; i < 8; i++)
    pinW(LCD_DB[i], (value >> i) & 1);
}

/* ============================================================
   Chip select logic
   ============================================================ */
static inline bool chipNeedsCS2Active(uint8_t chip)
{
#if LEFT_CHIP_IS_CS2
  return (chip == CHIP_LEFT);
#else
  return (chip == CHIP_RIGHT);
#endif
}

/* ============================================================
   Byte write
   ============================================================ */
static void lcdWriteByte(uint8_t chip, bool isData, uint8_t value)
{
  setA0(isData);       // CN10-4 = A0
  setDataBus(value);   // CN10-9..16 = D0..D7

  modeIdle();
  delayMicroseconds(1);

  modeWriteWindowBegin();                 // CN10-5 LOW only for write
  setCS2(chipNeedsCS2Active(chip));      // choose half

  delayMicroseconds(T_SETUP_US);
  pulseStrobe();                         // CN10-8 main strobe

  setCS2(false);
  modeWriteWindowEnd();

  delayMicroseconds(T_HOLD_US);
}

static inline void lcdCmd(uint8_t chip, uint8_t cmd)
{
  lcdWriteByte(chip, false, cmd);
}

static inline void lcdData(uint8_t chip, uint8_t data)
{
  lcdWriteByte(chip, true, data);
}

/* ============================================================
   Base commands
   ============================================================ */
static void lcdResetHw(void)
{
  modeIdle();
  setCS2(false);
  setA0(false);
  stbIdle();

  setRST(true);
  delay(30);
  setRST(false);
  delay(60);
}

static uint8_t chipBaseCol(uint8_t chip)
{
  return (chip == CHIP_LEFT) ? COL_START_LEFT : COL_START_RIGHT;
}

static void lcdSetPageCol(uint8_t chip, uint8_t page, uint8_t col)
{
  uint8_t phys = chipBaseCol(chip) + col;

  if (phys >= LCD_RAM_COLS)
    phys = LCD_RAM_COLS - 1;

  lcdCmd(chip, CMD_SET_PAGE | (page & 0x03));
  lcdCmd(chip, CMD_SET_COLUMN | phys);
}

static void lcdInitChip(uint8_t chip)
{
  lcdCmd(chip, CMD_DISPLAY_OFF);
  delay(4);

  lcdCmd(chip, CMD_RESET);
  delay(4);

  lcdCmd(chip, CMD_DUTY_16);
  delay(4);

#if LCD_ADC_REVERSED
  lcdCmd(chip, CMD_ADC_REVERSE);
#else
  lcdCmd(chip, CMD_ADC_FORWARD);
#endif
  delay(4);

  lcdCmd(chip, CMD_STATIC_DRIVE_OFF);
  delay(4);

  lcdCmd(chip, CMD_DISPLAY_START_LINE | 0);
  delay(4);

  lcdCmd(chip, CMD_DISPLAY_ON);
  delay(4);
}

static void lcdInit(void)
{
  lcdResetHw();
  lcdInitChip(CHIP_LEFT);
  lcdInitChip(CHIP_RIGHT);
}

/* ============================================================
   Buffer ops
   ============================================================ */
static void lcdBufClear(void)
{
  for (uint8_t p = 0; p < LCD_PAGES; p++)
    for (uint8_t x = 0; x < LCD_WIDTH; x++)
      lcd_fb[p][x] = 0x00;
}

static void lcdBufClearPage(uint8_t page)
{
  if (page >= LCD_PAGES) return;
  for (uint8_t x = 0; x < LCD_WIDTH; x++)
    lcd_fb[page][x] = 0x00;
}

static void lcdBufFill(uint8_t value)
{
  for (uint8_t p = 0; p < LCD_PAGES; p++)
    for (uint8_t x = 0; x < LCD_WIDTH; x++)
      lcd_fb[p][x] = value;
}

/*
  Update model:
    For each chip:
      For each page:
        Set Page
        Set Column(0)
        67 data bytes
        pause ~200 us
*/
static void lcdUpdate(void)
{
  // LEFT chip
  for (uint8_t page = 0; page < LCD_PAGES; page++)
  {
    lcdSetPageCol(CHIP_LEFT, page, 0);
    for (uint8_t x = 0; x < LCD_HALF_WIDTH; x++)
      lcdData(CHIP_LEFT, lcd_fb[page][x]);

    delayMicroseconds(PAGE_PAUSE_US);
  }

  // RIGHT chip
  for (uint8_t page = 0; page < LCD_PAGES; page++)
  {
    lcdSetPageCol(CHIP_RIGHT, page, 0);
    for (uint8_t x = 0; x < LCD_HALF_WIDTH; x++)
      lcdData(CHIP_RIGHT, lcd_fb[page][LCD_HALF_WIDTH + x]);

    delayMicroseconds(PAGE_PAUSE_US);
  }
}

static void lcdAllOn(void)
{
  lcdBufFill(0xFF);
  lcdUpdate();
}

static const uint8_t* lcdGlyph(char c)
{
  if (c >= 'a' && c <= 'z') c -= 32;

  switch (c)
  {
    case ' ': return GLYPH_SPACE;
    case '-': return GLYPH_DASH;
    case '.': return GLYPH_DOT;
    case ':': return GLYPH_COLON;
    case '/': return GLYPH_SLASH;

    case '0': return GLYPH_0;
    case '1': return GLYPH_1;
    case '2': return GLYPH_2;
    case '3': return GLYPH_3;
    case '4': return GLYPH_4;
    case '5': return GLYPH_5;
    case '6': return GLYPH_6;
    case '7': return GLYPH_7;
    case '8': return GLYPH_8;
    case '9': return GLYPH_9;

    case 'A': return GLYPH_A;
    case 'B': return GLYPH_B;
    case 'C': return GLYPH_C;
    case 'D': return GLYPH_D;
    case 'E': return GLYPH_E;
    case 'F': return GLYPH_F;
    case 'G': return GLYPH_G;
    case 'H': return GLYPH_H;
    case 'I': return GLYPH_I;
    case 'J': return GLYPH_J;
    case 'K': return GLYPH_K;
    case 'L': return GLYPH_L;
    case 'M': return GLYPH_M;
    case 'N': return GLYPH_N;
    case 'O': return GLYPH_O;
    case 'P': return GLYPH_P;
    case 'Q': return GLYPH_Q;
    case 'R': return GLYPH_R;
    case 'S': return GLYPH_S;
    case 'T': return GLYPH_T;
    case 'U': return GLYPH_U;
    case 'V': return GLYPH_V;
    case 'W': return GLYPH_W;
    case 'X': return GLYPH_X;
    case 'Y': return GLYPH_Y;
    case 'Z': return GLYPH_Z;
  }

  return GLYPH_SPACE;
}

static uint8_t lcdOrientGlyphByte(uint8_t b)
{
#if FONT_LSB_TOP
  return b;
#else
  uint8_t r = 0;
  for (uint8_t i = 0; i < 7; i++)
    if (b & (1 << i))
      r |= (1 << (6 - i));
  return r;
#endif
}

/* ============================================================
   Text
   ============================================================ */
static void lcdBufPrintLine(uint8_t page, const char* s)
{
  if (page >= LCD_PAGES) return;

  lcdBufClearPage(page);

  uint8_t x = 0;
  uint8_t chars = 0;

  while (*s && chars < LCD_CHARS_PER_LINE && x < LCD_WIDTH)
  {
    const uint8_t* g = lcdGlyph(*s);

    for (uint8_t i = 0; i < 6; i++)
    {
      uint8_t b = 0x00;
      if (i < 5) b = pgm_read_byte(g + i);
      b = lcdOrientGlyphByte(b);

      if (x >= LCD_WIDTH) break;
      lcd_fb[page][x++] = b;
    }

    s++;
    chars++;
  }
}

static void lcdShowFullText(const char* line1, const char* line2)
{
  lcdBufClear();
  lcdBufPrintLine(0, line1);
  lcdBufPrintLine(1, line2);
  lcdUpdate();
}

/* ============================================================
   Arduino
   ============================================================ */
void setup()
{
  Serial.begin(115200);

  pinMode(LCD_PIN_A0, OUTPUT);
  pinMode(LCD_PIN_MODE, OUTPUT);
  pinMode(LCD_PIN_STB, OUTPUT);
  pinMode(LCD_PIN_CS2, OUTPUT);
  pinMode(LCD_PIN_RST, OUTPUT);

  for (uint8_t i = 0; i < 8; i++)
  {
    pinMode(LCD_DB[i], OUTPUT);
    digitalWrite(LCD_DB[i], LOW);
  }

  modeIdle();
  setCS2(false);
  setA0(false);
  stbIdle();
  setRST(false);

  delay(50);

  lcdInit();

  lcdAllOn();
  delay(1000);

  lcdShowFullText(
    "1234567890123456789012",
    "ABCDEFGHIJKLMNOPQRSTUV"
  );
}
