/*
LCD 134x16, SED1520-like
Подключение:
D2 -> CN10-9 (DB0)
D3 -> CN10-10 (DB1)
D4 -> CN10-11 (DB2)
D5 -> CN10-12 (DB3)
D6 -> CN10-13 (DB4)
D7 -> CN10-14 (DB5)
D8 -> CN10-15 (DB6)
D9 -> CN10-16 (DB7)

text

D10 -> CN10-4   (A0)
D11 -> CN10-5   (/WR)
D12 -> CN10-8   (/WR1)
A1  -> CN10-18  (/CS2)
A2  -> CN10-17  (RST)
*/

/* ============================================================
Команды SED1520
============================================================ */
#define CMD_DISPLAY_OFF 0xAE
#define CMD_DISPLAY_ON 0xAF
#define CMD_DISPLAY_START_LINE 0xC0 // + 0..31
#define CMD_SET_PAGE 0xB8 // + 0..3
#define CMD_SET_COLUMN 0x00 // + 0..79
#define CMD_ADC_FORWARD 0xA0
#define CMD_ADC_REVERSE 0xA1
#define CMD_STATIC_DRIVE_OFF 0xA4
#define CMD_STATIC_DRIVE_ON 0xA5
#define CMD_DUTY_16 0xA8
#define CMD_DUTY_32 0xA9
#define CMD_RMW 0xE0
#define CMD_END 0xEE
#define CMD_RESET 0xE2

/* ============================================================
Пины
============================================================ */
const uint8_t LCD_DB[8] = {2,3,4,5,6,7,8,9};

#define LCD_PIN_A0 10 // CN10-4
#define LCD_PIN_WR 11 // CN10-5
#define LCD_PIN_WR1 12 // CN10-8
#define LCD_PIN_CS2 A1 // CN10-18
#define LCD_PIN_RST A2 // CN10-17

/* ============================================================
Геометрия
============================================================ */
#define LCD_WIDTH 134
#define LCD_HEIGHT 16
#define LCD_PAGES 2
#define LCD_HALF_WIDTH 67
#define LCD_COLS_PER_IC 80

// По VCD H8 выставляет колонку с 0
#define COL_START0 0
#define COL_START1 0

/* ============================================================
Настройки
============================================================ */
#define RST_ACTIVE_HIGH 1
#define A0_DATA_HIGH 1
#define WR_ACTIVE_LOW 1
#define CS2_ACTIVE_LOW 1

// Если окажется зеркально по X -> 1
#define LCD_ADC_REVERSED 0

// Если символы внутри строки "вверх ногами" -> 0
#define FONT_LSB_TOP 1

/* ============================================================
Тайминги
============================================================ */
#define T_SETUP_US 4
#define T_PULSE_US 6
#define T_HOLD_US 6

/* ============================================================
Буфер [page][x]
============================================================ */
static uint8_t lcd_fb[LCD_PAGES][LCD_WIDTH];

/* ============================================================
Низкий уровень
============================================================ */
static inline void lcdPinWrite(uint8_t pin, bool level)
{
digitalWrite(pin, level ? HIGH : LOW);
}

static inline bool lcdActiveToLevel(bool active, bool activeLow)
{
return activeLow ? !active : active;
}

static inline void lcdSetA0(bool isData)
{
lcdPinWrite(LCD_PIN_A0, A0_DATA_HIGH ? isData : !isData);
}

static inline void lcdSetCS2(bool active)
{
lcdPinWrite(LCD_PIN_CS2, lcdActiveToLevel(active, CS2_ACTIVE_LOW));
}

static inline void lcdSetRST(bool active)
{
lcdPinWrite(LCD_PIN_RST, RST_ACTIVE_HIGH ? active : !active);
}

static inline void lcdWRIdle(void)
{
lcdPinWrite(LCD_PIN_WR, lcdActiveToLevel(false, WR_ACTIVE_LOW));
}

static inline void lcdWR1Idle(void)
{
lcdPinWrite(LCD_PIN_WR1, lcdActiveToLevel(false, WR_ACTIVE_LOW));
}

static inline void lcdPulseWR(void)
{
lcdPinWrite(LCD_PIN_WR, lcdActiveToLevel(true, WR_ACTIVE_LOW));
delayMicroseconds(T_PULSE_US);
lcdPinWrite(LCD_PIN_WR, lcdActiveToLevel(false, WR_ACTIVE_LOW));
}

static inline void lcdPulseWR1(void)
{
lcdPinWrite(LCD_PIN_WR1, lcdActiveToLevel(true, WR_ACTIVE_LOW));
delayMicroseconds(T_PULSE_US);
lcdPinWrite(LCD_PIN_WR1, lcdActiveToLevel(false, WR_ACTIVE_LOW));
}

static void lcdSetDataBus(uint8_t value)
{
for (uint8_t i = 0; i < 8; i++)
lcdPinWrite(LCD_DB[i], (value >> i) & 1);
}

/* ============================================================
Запись
============================================================ */

// Левая половина: через WR1
static void lcdWriteHalf0(bool isData, uint8_t value)
{
lcdSetA0(isData);
lcdSetDataBus(value);
lcdSetCS2(false);

delayMicroseconds(T_SETUP_US);
lcdPulseWR1();
delayMicroseconds(T_HOLD_US);
}

// Правая половина: через CS2 + WR
static void lcdWriteHalf1(bool isData, uint8_t value)
{
lcdSetA0(isData);
lcdSetDataBus(value);

delayMicroseconds(T_SETUP_US);
lcdSetCS2(true);
lcdPulseWR();
lcdSetCS2(false);
delayMicroseconds(T_HOLD_US);
}

static inline void lcdWrite(uint8_t half, bool isData, uint8_t value)
{
if (half == 0) lcdWriteHalf0(isData, value);
else lcdWriteHalf1(isData, value);
}

static inline void lcdCmd(uint8_t half, uint8_t cmd)
{
lcdWrite(half, false, cmd);
}

static inline void lcdData(uint8_t half, uint8_t data)
{
lcdWrite(half, true, data);
}

/* ============================================================
Инициализация
============================================================ */
static void lcdResetHw(void)
{
lcdWRIdle();
lcdWR1Idle();
lcdSetCS2(false);
lcdSetA0(false);

lcdSetRST(true);
delay(30);
lcdSetRST(false);
delay(50);
}

static void lcdSetPageCol(uint8_t half, uint8_t page, uint8_t col)
{
uint8_t base = (half == 0) ? COL_START0 : COL_START1;
uint8_t phys = base + col;

if (phys >= LCD_COLS_PER_IC) return;

lcdCmd(half, CMD_SET_PAGE | (page & 0x03));
lcdCmd(half, CMD_SET_COLUMN | phys); // ПРАВИЛЬНО: без &0x4F
}

static void lcdInitHalf(uint8_t half)
{
lcdCmd(half, CMD_DISPLAY_OFF);
delay(2);

lcdCmd(half, CMD_RESET);
delay(2);

lcdCmd(half, CMD_DUTY_16);
delay(2);

#if LCD_ADC_REVERSED
lcdCmd(half, CMD_ADC_REVERSE);
#else
lcdCmd(half, CMD_ADC_FORWARD);
#endif
delay(2);

lcdCmd(half, CMD_STATIC_DRIVE_OFF);
delay(2);

lcdCmd(half, CMD_DISPLAY_START_LINE | 0);
delay(2);

lcdCmd(half, CMD_END);
delay(2);

lcdCmd(half, CMD_DISPLAY_ON);
delay(2);
}

static void lcdInit(void)
{
lcdResetHw();
lcdInitHalf(0);
lcdInitHalf(1);
}

/* ============================================================
Буфер
============================================================ */
static void lcdBufClear(void)
{
for (uint8_t p = 0; p < LCD_PAGES; p++)
for (uint8_t x = 0; x < LCD_WIDTH; x++)
lcd_fb[p][x] = 0x00;
}

static void lcdBufFill(uint8_t p0, uint8_t p1)
{
for (uint8_t x = 0; x < LCD_WIDTH; x++)
{
lcd_fb[0][x] = p0;
lcd_fb[1][x] = p1;
}
}

static void lcdUpdate(void)
{
for (uint8_t page = 0; page < LCD_PAGES; page++)
{
lcdSetPageCol(0, page, 0);
for (uint8_t x = 0; x < LCD_HALF_WIDTH; x++)
lcdData(0, lcd_fb[page][x]);

text

lcdSetPageCol(1, page, 0);
for (uint8_t x = 0; x < LCD_HALF_WIDTH; x++)
  lcdData(1, lcd_fb[page][LCD_HALF_WIDTH + x]);
}
}

static void lcdClear(void)
{
lcdBufClear();
lcdUpdate();
}

static void lcdAllOn(void)
{
lcdBufFill(0xFF, 0xFF);
lcdUpdate();
}

static const uint8_t* lcdGlyph(char c)
{
if (c >= 'a' && c <= 'z') c -= 32;

switch (c)
{
case ' ': return GLYPH_SPACE;
case '-': return GLYPH_DASH;
case '.': return GLYPH_DOT;
case ':': return GLYPH_COLON;
case '/': return GLYPH_SLASH;

text

case '1': return GLYPH_1;
case '2': return GLYPH_2;
case '3': return GLYPH_3;

static uint8_t lcdOrientGlyphByte(uint8_t b)
{
#if FONT_LSB_TOP
return b;
#else
uint8_t r = 0;
for (uint8_t i = 0; i < 7; i++)
if (b & (1 << i))
r |= (1 << (6 - i));
return r;
#endif
}

/* ============================================================
Текст в буфер
============================================================ /
static void lcdBufPrintPage(uint8_t page, uint8_t x, const char s)
{
if (page >= LCD_PAGES || x >= LCD_WIDTH) return;

while (s && x < LCD_WIDTH)
{
const uint8_t g = lcdGlyph(*s);

text

for (uint8_t i = 0; i < 6; i++)
{
  uint8_t b = 0x00;
  if (i < 5) b = pgm_read_byte(g + i);
  b = lcdOrientGlyphByte(b);

  if (x >= LCD_WIDTH) break;
  lcd_fb[page][x++] = b;
}

s++;
}
}

static void lcdShowText(const char* line1, const char* line2)
{
lcdBufClear();
lcdBufPrintPage(0, 0, line1);
lcdBufPrintPage(1, 0, line2);
lcdUpdate();
}

void setup()
{
Serial.begin(115200);

pinMode(LCD_PIN_A0, OUTPUT);
pinMode(LCD_PIN_WR, OUTPUT);
pinMode(LCD_PIN_WR1, OUTPUT);
pinMode(LCD_PIN_CS2, OUTPUT);
pinMode(LCD_PIN_RST, OUTPUT);

for (uint8_t i = 0; i < 8; i++)
{
pinMode(LCD_DB[i], OUTPUT);
digitalWrite(LCD_DB[i], LOW);
}

lcdWRIdle();
lcdWR1Idle();
lcdSetCS2(false);
lcdSetA0(false);
lcdSetRST(false);

delay(50);

lcdInit();

// Проверка: весь экран
lcdAllOn();
delay(1000);

// Очистка + текст
lcdShowText("HELLO", "LCD TEST");
}
